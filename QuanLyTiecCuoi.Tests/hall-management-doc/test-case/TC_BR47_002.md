### Selection Handler

```csharp
public Hall SelectedItem
{
    set
    {
        _selectedItem = value;
        if (value != null && IsEditing)
        {
            HallName = value.HallName;
            SelectedHallType = HallTypes.FirstOrDefault(t => t.HallTypeId == value.HallTypeId);
            MaxTableCount = value.MaxTableCount;
            Note = value.Note;
        }
        OnPropertyChanged();
    }
}
```
