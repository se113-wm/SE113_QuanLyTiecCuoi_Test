### Service Method

```csharp
public async Task<bool> Update(HallDTO hallDto)
{
    var hall = await _context.Halls.FindAsync(hallDto.HallId);
    if (hall == null) return false;

    hall.HallName = hallDto.HallName;
    hall.HallTypeId = hallDto.HallTypeId;
    hall.MaxTableCount = hallDto.MaxTableCount;
    hall.Note = hallDto.Note;

    await _context.SaveChangesAsync();
    return true;
}
```
