### DAO Create Method

```csharp
public class ShiftDAO
{
    public void Create(ShiftDTO shift)
    {
        var entity = new Shift
        {
            ShiftName = shift.ShiftName,
            StartTime = shift.StartTime,
            EndTime = shift.EndTime
        };
        _context.Shifts.Add(entity);
        _context.SaveChanges();
    }
}
```
