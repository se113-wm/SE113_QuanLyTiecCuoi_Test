### Change Detection

```csharp
private bool HasChanges()
{
    var original = OriginalList.First(s => s.ShiftId == SelectedItem.ShiftId);
    return SelectedItem.ShiftName != original.ShiftName ||
           SelectedItem.StartTime != original.StartTime ||
           SelectedItem.EndTime != original.EndTime;
}

if (!HasChanges())
{
    ShowMessage("MSG16", "Không có thay đổi nào để cập nhật");
    return;
}
```
