### DAO Delete Method

```csharp
public class ShiftDAO
{
    public void Delete(int shiftId)
    {
        var shift = _context.Shifts.Find(shiftId);
        if (shift != null)
        {
            _context.Shifts.Remove(shift);
            _context.SaveChanges();
        }
    }
}
```
